## 如何理解代理模式
1. 一个真实对象
2. 一个代理对象
3. 一个接口：`代理对象`拥有`真实对象`的属性/方法 —— `接口`。可通过 `接口` 操作对`真实对象`添加额外的处理


## 应用场景 —— 异步

### Loading
```js
// 原对象
var Mod = {
    init: function (data) {
        // 主逻辑
    }
}
// 代理对象
var ProxyLoading = {
    // 代理对象接口
    init: function (mod) {
        showLoading();

        if (请求成功) {
            hideLoading();
            mod.init(data);
        } else {
            hideLoading();
        }
    }
｝

// 调用接口
ProxyLoading.init(Mod);
```


### 图片懒加载
```js
// 原对象
var myImage = (function(){
    var imgNode = document.createElement('img');
    document.body.appendChild(imgNode);
    return {
        setSrc: function(src) {
            imgNode.src = src;
        }
    }
})();

// 代理
var proxyImage = (function(){
    var img = new Image;
    img.onload = function() {
        myImage.setSrc(this.src);             // 图片加载完设置真实图片src
    }
    return {
        setSrc: function(src) {
            myImage.setSrc('./loading.gif');  // 预先设置图片src为loading图
            img.src = src;
        }
    }
})();

// 外部调用
proxyImage.setSrc('./product.png');           // 有loading图的图片预加载效果
```

### ES6的Proxy实现图片懒加载
```js
const createImgProxy = (img, loadingImg, realImg) => {
  let hasLoaded = false;

  // 原对象处理 - 异步
  const virtualImg = new Image();
  virtualImg.src = realImg;
  virtualImg.onload = () => {
    Reflect.set(img, 'src', realImg);
    hasLoaded = true;
  }

  // 代理对象处理
  return new Proxy(img, {
    get(obj, prop) {
      if (prop === 'src' && !hasLoaded) {
        return loadingImg;
      }
      return obj[prop];
    }
  });
};
const img = new Image();
const imgProxy = createImgProxy(img, '/loading.gif', '/some/big/size/img.jpg');
document.body.appendChild(imgProxy);
```
