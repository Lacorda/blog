## 对字符串、数组、对象、函数参数的解析赋值

## 用法示例
```js
const arr = [1, 2, 3, 4, 5];
const obj = { a: 'a', b: 'b', c: 'c', d: 'd', e: 'e' };

// 取数组元素 或 对象属性值
let [ a, b ] = arr; // a = 1; b = 2
let { a, b } = obj;  // a = 'a'; b = 'b'

// 还可以设置默认值，当当前值为undefined时取默认值
let [a = 11, b, c, d, e, f = 123] = arr; // a = 1; f = 123, arr不变
let { a = 'aa', f = 'fff'} = obj; // a = 'a'; f = 'fff'

// 支持多层级
const arr2 = [1,[2,3], [4, [5], 6]];
const obj2 = {
    a: 'a',
    b: {
        b: 'b',
        c: 'c',
    },
    c: {
        d: 'd',
        e: { e: 'e' },
        f: 'f'
    }
};
let [, [a]] = arr2; // a = 2;
let { b: { c }} = obj2; // c = 'c';

// 数组/对象对参数时，是一样的

// 对字符串的解构赋值
const [a, b, c, d, e] = 'hello';
a // "h"
b // "e"
c // "l"
d // "l"
e // "o"
```
