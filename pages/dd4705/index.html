webpack转vite

# 为什么Vite这么快？

  • 底层实现不同：webpack使用node.js构建；vite使用esbuild预构建，esbuild是使用go编写的
  • 启动方式不同：webpack是先打包，再启动开发服务器；vite是基于现代浏览器支持ES Module，不需要打包，直接启动开发服务器，再实时编译（请求了哪个模块才对该模块进行编译）
  • 打包方式不同：vite使用rollup打包

# 6.打包时移除cdn引入的包

import commonjs from '@rollup/plugin-commonjs';
import externalGlobals from 'rollup-plugin-external-globals';
const externals = {
  vue: 'Vue',
  'vue-router': 'VueRouter',
  axios: 'axios',
  vuex: 'Vuex',
  'element-ui': 'ELEMENT',
  lodash: '_',
  echarts: 'echarts',
  'v-charts': 'VeIndex',
  qs: 'Qs',
};

export default defineConfig({
  build: {
    rollupOptions: {
      external: ['vue', 'vue-router', 'axios', 'vuex', 'element-ui', 'lodash', 'echarts', 'v-charts', 'qs'],
      plugins: [
        commonjs(), // 转换 CJS -> ESM, 通过 cdn 引入的没法转
        externalGlobals(externals),
      ],
    },
  },
});

# 7.修改require.context

改成`import.meta.globEager`或`import.meta.glob`

let routes = [];
// 方式一
const files = import.meta.globEager('./modules/*.js');
for (const key in files) {
  const fileData = files[key].default;
  routes[fileData.index] = fileData.routers;
}
// 方式二
const files2 = import.meta.glob('./modules/*.js');
for (const key in files2) {
  files2[key]().then(res => {
    const fileData = res.default;
    routes[fileData.index] = fileData.routers;
  });
}

# 8.打包压缩

使用`vite-plugin-compression`

import compressPlugin from "vite-plugin-compression";

export default defineConfig({
  plugins: [
    compressPlugin({
      filter: /\.(js|css)$/i, // 压缩文件类型
      verbose: true,
      disable: false,
      threshold: 10240,
      algorithm: 'gzip',
      ext: '.gz',
    }),
  ],
});

  • Vite是很快，效率更高
  • 缺点：只能针对现代浏览器；commonjs模块不兼容；社区生态还在完善
