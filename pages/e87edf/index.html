Promise和async-await

# 概念

# 同步与异步

  • 同步：会逐行执行代码，会对后续代码造成阻塞，直至代码接收到预期的结果之后，才会继续向下执行任务
  • 异步：调用之后先不管结果，继续向下执行任务，常见的有：setTimeout, promise, 事件回调等

# Promise

一种异步处理的代码规范

# Promise的使用和状态

# 语法

  • Promise对象的回调函数，带两个参数resolve和reject
  • resolve()方法表示符合预期，参数为预期返回值，会作为then()方法的入参
  • reject()方法表示不符合预期，参数会作为catch()方法的入参
  • Promise链式写法：因为then()/ catch()方法返回的还是Promise
  • 快捷写法
Promise.resolve(1);
// 等价于
new Promise((resolve) => resolve(1));

Promise.reject(new Error("BOOM"))
// 等价于
new Promise((resolve, reject) => reject(new Error("BOOM")));

# 三种状态

  • Pending准备态
  • onFulfilled完成态
  • onRejected拒绝态

onFulfilled和onRejected都是最终态，最终态不可逆

# Promise.all

  • Promise.all()方法用于将多个 Promise 实例，包装成一个新的 Promise 实例
  • 全部Promise状态都为onFulfilled时，才会执行then()，否则执行catch()
  • then()方法返回结果是个数组，元素为每个Promise的返回值
const p = Promise.all([p1, p2, p3]);
1

# Promise.race

  • Promise.race()方法用于将多个 Promise 实例，包装成一个新的 Promise 实例
  • 只要有一个Promise状态变成onFulfilled，才会执行then()
  • then()方法返回结果为最先变成onFulfilled态的的返回值

# async...await

  • 一个ES6异步写法的语法糖
  • async函数返回一个Promise对象
  • await后跟一个Promise对象，返回该Promise对象的返回值
  • 没有await时，单独async函数相当于是将函数变为一个Promise
  • 有await时，相当于是将Promise异步改成同步，需等待Promise有返回值时，才继续执行

# 手动实现promise对象功能

// 第一版：promise()方法执行同步内容，.then执行异步函数
function P(fn) {
    let callback = null;

    this.then = function (cb) { // 执行异步函数
        callback = cb;
    }

    function resolve(value) {  // 假的异步函数，无状态
        setTimeout(() => {
            callback(value); // 给callback传值
        }, 0)
    }

    fn(resolve);  // 同步内容
}
const doSomething = function () {
    return new P(resolve => {
        console.log('同步内容');
        resolve('异步返回');
    })
}
// 只执行doSomething时，resolve里的callback为null，会报错。
// 因为这边没有添加状态，导致resolve未到then就被执行了
// 此处返回：'同步内容'  '异步返回'
doSomething().then(res => {
    console.log(res);
});

// 第二版：添加状态resolve是否可执行
function P(fn) {
    let state = 'pending';
    let value;
    let deferred; // 用于控制resolved状态

    function handler(onResolved) {
		if (state === 'pending') {
			deferred = onResolved;
			return;
		}

		onResolved(value);
    }

    function resolve(resolveValue) {
        value = resolveValue;
        state = 'resolved';

        if (deferred) { // 用deferred控制是否可执行回调
            handler(deferred);
        }
    }

    this.then = function (cb) { // 执行异步函数
        handler(cb)
    }

    fn(resolve);  // 同步内容
}
const doSomething = function () {
    return new P(resolve => {
        console.log('同步内容');
        resolve('异步返回');
    })
}
const p = doSomething(); // 此处返回：'异步返回'
p.then(res => {
    console.log(res); // 此处返回：'异步返回'
})

// 第三版：给then添加链式调用
function P(fn) {
    let state = 'pending';
    let value;
    let deferred; // 用于控制resolved状态

    function handler({onResolved,resolve}) { // onResolved: 上一个then的回调
		if (state === 'pending') {
			deferred = {onResolved,resolve};
			return;
		}

        if (!onResolved) { // 没有上一个then的回调，直接执行当前resolve
            resolve(value);
			return;
        }

		let ret = onResolved(value); // 上一个then的回调的返回值，作为下一个then的入参
        resolve(ret);
    }

    function resolve(resolveValue) {
        if (resolveValue && typeof resolveValue.then === 'function') {
            // 当上一个then返回的是一个P时，递归执行resolve方法直至解析出值出来
            resolveValue.then(resolve);
            return;
        }

        value = resolveValue;
        state = 'resolved';

        if (deferred) { // 用deferred控制是否可执行回调
            handler(deferred);
        }
    }

    this.then = function (onResolved) {
        return new P(resolve => {
            handler({
                onResolved, // 上一个then的回调
                resolve, // 当前then的回调
            })
        })
    }

    fn(resolve);  // 同步内容
}
const doSomething = function () {
    return new P(resolve => {
        console.log('同步内容');
        resolve('异步返回');
    })
}
const p = doSomething(); // 此处返回：'同步内容'
p.then(res => {
    console.log(res); // 此处返回：'异步返回'
    return '1 then'
}).then(res => {
    console.log('2: ', res); // 此处返回：'2: 1 then'
    return '2 then'
}).then(res => {
    console.log('3: ', res); // 此处返回：'3: 3 then'
    return '3 then'
})

// 第四版：添加reject状态，保证resolve和reject都是异步的
function P(fn) {
    let state = 'pending';
    let value;
    let deferred; // 用于控制resolved状态

    function handler({onResolved, resolve, onRejected, reject}) { // onResolved: 上一个then的回调
		if (state === 'pending') {
			deferred = {onResolved,resolve};
			return;
		}

        setTimeout(() => { // 利用setTimeout保证resolve和reject都是异步的
            let handlerCb;

            if (state === 'resolved') {
                handlerCb = onResolved;
            } else {
                handlerCb = onRejected;
            }

            if (!handlerCb) { // 没有上一个then的回调，直接执行当前resolve
                if (state === 'resolved') {
                    resolve(value)
                } else {
                    reject(value)
                }
                return;
            }

            let ret
            try {
                ret = handlerCb(value); // 上一个then的回调的返回值，作为下一个then的入参
            } catch (e) {
                reject(e); // 上一个then抛出异常时，直接reject
                return;
            }
            resolve(ret);
        }, 0);
    }

    function resolve(resolveValue) {
        if (resolveValue && typeof resolveValue.then === 'function') {
            // 当上一个then返回的是一个P时，递归执行resolve方法直至解析出值出来
            resolveValue.then(resolve);
            return;
        }

        value = resolveValue;
        state = 'resolved';

        if (deferred) { // 用deferred控制是否可执行回调
            handler(deferred);
        }
    }

    function reject(rejectValue) {
        state = 'rejected';
        value = rejectValue;

        if (deferred) {
            handle(deferred);
        }
    }

    this.then = function (onResolved, onRejected) {
        return new P((resolve, reject) => {
            handler({
                onResolved, // 上一个then的回调
                resolve, // 当前then的回调
                onRejected,
                reject,
            })
        })
    }

    fn(resolve);  // 同步内容
}
const doSomething = function (i) {
    return new P((resolve, reject) => {
        console.log('同步内容');
        if (i < 3) {
            resolve('异步返回 resolve: ', i);
        } else {
            reject('异步返回 reject: ', i);
        }
    })
}
const p = doSomething(2);  // 此处返回：'同步内容'
p.then(res => {
    console.log(res); // 此处返回：'异步返回 resolve: 2'
}, err => {
    console.log(err);
})
const p1 = doSomething(4);  // 此处返回：'同步内容'
p1.then(res => {
    console.log(res);
}, err => {
    console.log(err); // 此处返回：'异步返回 reject: 4'
})

# 手写一个超时器：利用promise最终态不可逆的原理

const timeouter = (promise, delay) => {
    return new Promise((resolve, reject) => {
        const timeout = setTimeout(() => {
            reject(new Error('time out'));
        }, delay);
        promise.then((res) => {
            clearTimeout(timeout);
            resolve(res);
        }, reject)
    })
}
// 3秒未返回，则报错'time out'
timeout(promise, 3000).then(res => {
    console.log(res); // 返回接口数据
}).catch(err => {
    console.log(err); // 返回接口报错 或者 超过三秒返回'time out'
})

# 红灯三秒亮一次，绿灯一秒亮一次，黄灯2秒亮一次；如何让三个灯不断交替重复亮灯？

function red() {
    console.log('red: ', Date.now());
}
function green() {
    console.log('green: ', Date.now());
}
function yellow() {
    console.log('yellow: ', Date.now());
}

// 使用setTimeout
function loop () {
    setTimeout(() => {
        red();
        setTimeout(() => {
            green();

            setTimeout(() => {
                yellow();

            }, 2000)
        }, 1000)
    }, 3000)
}
loop();
setInterval(loop, 6000);

// 使用Promise: 利用Promise的链式调用 + 递归
function tic(cb, delay) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            cb();
            resolve();
        }, delay)
    })
}
function loop() {
    Promise.resolve().then(() => {
        return tic(red, 3000)
    }).then(() => {
        return tic(green, 1000)
    }).then(() => {
        return tic(yellow, 2000)
    }).then(() => {
        loop();
    })
}
loop();

// 使用Generator: 利用next和done判断做递归
function* show() {
    yield tic(red, 3000);
    yield tic(green, 1000);
    yield tic(yellow, 2000);
}
function loop(iterator) {
    const res = iterator.next();
    if (res.done) {
        loop(show());
    } else {
        res.value.then(() => {
            loop(iterator); // 保证next的是同一个iterator
        })
    }
}
loop(show());

// 使用await... async  + while
(async function(){
    while(true){
        await tic(red, 3000);
        await tic(green, 1000);
        await tic(yellow, 2000);
    }
})();
